 ## Config for SKR Mini E3 V2.0 Rev1
    ################################
    ####  Simple Screen Config  ####
    ################################
    [display]
    lcd_type: sh1106
    click_pin: ^!PA15
    i2c_bus: i2c1a
    i2c_mcu = mcu
    encoder_pins: ^PA1, ^PC14
    kill_pin: ^!PB15
    vcomh: 60
    x_offset: 2
    
    [neopixel display_led]
    pin: PA8
    color_order: GRB
    initial_RED: 0.5
    initial_GREEN: 0.5
    initial_BLUE: 0.5
    
    #sda = pb9
    #scl = pb8
